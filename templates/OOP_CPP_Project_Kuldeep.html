void emp::search() {

    string str, s;

    cout<<"\nEnter something to search : ";
    cin>>s;

    ifstream info("info.txt");

    while (!info.eof())
    {
        getline(info,str);

        bool match;
        for (int i = 0; i < str.size(); i++)
        {
            match = false;

            if(str[i]==s[0]){
                for (int j = 0; j < s.size(); j++)
                {
                    if(str[i+j]==s[j]){
                        match = true;
                    }
                    else
                    {
                        match=false;
                        break;
                    }
                }
            }
            if(match==true){
                cout<<str<<"\n";
                break;
            }
        }
    }

    info.close();
}
