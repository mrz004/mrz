class product
{
public:
    int id, price;
    string name;

    product(){

        cin>>name;
        cout<<"\nEnter the id of product : ";
        cin>>id;
        cout<<"\nEnter the price of product : ";
        cin>>price;

    product(int a, string b, int c){
        id = a;
        name = b;
        price = c;
    }
    
    product(product &a){
        id = a.id;
        name = a.name;
        price = a.price;        
    }

    void display(){
        cout<<"\nProduct ID : "<<id<<"\nProduct name : "<<name<<"\nProduct price : "<<price;
    }
};

int main()
{
    int x;
    int y;
    string z;

    product p1;
    product p2 = p1;

    cin>>z;
    cout<<"\nEnter the id of product : ";
    cin>>x;
    cout<<"\nEnter the price of product : ";
    cin>>y;

    p1.display();
