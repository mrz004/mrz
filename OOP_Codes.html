Q.3) 

                #include<iostream>
                using namespace std;
                
                int main()
                {
                    int num, temp, rev=0;
                    cout<<"Enter the number : ";
                    cin>>num;
                    temp=num;
                    while (num>0)
                    {
                        rev=(rev*10)+(num%10);
                        num=num/10;
                    }
                    if (rev==temp)
                    {
                        cout<<"The number is a pelindrome.";
                    }
                    else
                    {
                        cout<<"The number is NOT a pelindrome.";
                    }
                
                    return 0;
                }
